import java.time.Duration;
import org.openqa.selenium.Alert;

public class CLASS_NAME {
    // Example: public class student_test

    WebDriver driver;

    @BeforeClass
    public void setup() {

        System.setProperty(
                "webdriver.chrome.driver",
                "CHROMEDRIVER_PATH");
        // Example: "C:\\Drivers\\chromedriver.exe"
        // Note: Use double backslash \\ for Windows paths

        driver = new ChromeDriver();

        driver.manage().window().maximize();

        driver.manage().timeouts()
                .pageLoadTimeout(Duration.ofSeconds(10));

        driver.get("BASE_URL");
        // Example: "http://localhost/student/admin"
    }

    @Test(priority = 1)
    public void adminLogin() throws InterruptedException {

        Thread.sleep(2000);

        WebElement username =
                driver.findElement(By.name("USERNAME_FIELD_NAME"));
        // Example: By.name("username")

        WebElement password =
                driver.findElement(By.name("PASSWORD_FIELD_NAME"));
        // Example: By.name("password")

        if (username.getAttribute("value").isEmpty()) {
            System.out.println("USERNAME_FIELD_NAME is empty");
            // Example: "username is empty"
        }

        if (password.getAttribute("value").isEmpty()) {
            System.out.println("PASSWORD_FIELD_NAME is empty");
            // Example: "password is empty"
        }

        username.sendKeys("ADMIN_USERNAME");
        // Example: "admin"

        password.sendKeys("ADMIN_PASSWORD");
        // Example: "admin@123"

        driver.findElement(
                By.xpath("LOGIN_BUTTON_XPATH"))
                .click();
        // Example: "//*[@id=\"login-frm\"]/button[1]"

        Thread.sleep(3000);
    }

    @Test(priority = 2)
    public void urlValidation() throws InterruptedException {

        driver.findElement(
                By.xpath("NAV_MENU_ITEM_XPATH"))
                .click();
        // Example: "//*[@id=\"Employee\"]"
        // Tip: inspect the sidebar/nav link and copy XPath

        String currentURL =
                driver.getCurrentUrl();

        String expectedURL =
                "EXPECTED_URL_AFTER_NAVIGATION";
        // Example: "http://localhost/student/admin/?page=students"

        if (currentURL.equals(expectedURL)) {

            System.out.println(
                    "URL Validation Success");

        } else {

            System.out.println(
                    "URL Validation Fail");
        }

        Thread.sleep(2000);
    }
