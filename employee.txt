import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.testng.annotations.AfterTest;
import org.testng.annotations.BeforeTest;
import org.testng.annotations.Test;

public class EmployeeTest {

    WebDriver driver;

    @BeforeTest
    public void setup() {

        System.setProperty("webdriver.chrome.driver",
                "C:\\chromedriver\\chromedriver.exe");

        driver = new ChromeDriver();
        driver.manage().window().maximize();

        // Load users.php page
        driver.get("http://localhost/employee/users.php");
    }

    @Test(priority = 1)
    public void loginTest() throws InterruptedException {

        driver.findElement(By.name("username"))
                .sendKeys("admin");

        driver.findElement(By.name("password"))
                .sendKeys("admin123");

        driver.findElement(By.name("login"))
                .click();

        Thread.sleep(2000);

        // Error message check
        if(driver.getPageSource().contains("Invalid")) {
            System.out.println("Error Message Displayed");
        }
        else {
            System.out.println("Login Successful");
        }
    }

    @Test(priority = 2, dependsOnMethods = "loginTest")
    public void searchEmployee() throws InterruptedException {

        WebElement search =
                driver.findElement(By.name("search"));

        search.clear();
        search.sendKeys("ab");

        Thread.sleep(2000);

        if(driver.getPageSource().contains("ab")) {
            System.out.println("data detected");
        }
        else {
            System.out.println("data undetected");
        }
    }

    @Test(priority = 3, dependsOnMethods = "searchEmployee")
    public void updateUser() throws InterruptedException {

        driver.findElement(By.xpath("//button[contains(text(),'Update')]"))
                .click();

        Thread.sleep(2000);

        // Close popup
        driver.findElement(By.xpath("//button[contains(text(),'Close')]"))
                .click();

        System.out.println("Update popup closed");
    }

    @Test(priority = 4, dependsOnMethods = "updateUser")
    public void addNewUser() throws InterruptedException {

        driver.findElement(By.xpath("//button[contains(text(),'Add New')]"))
                .click();

        Thread.sleep(2000);

        // Close popup
        driver.findElement(By.xpath("//button[contains(text(),'Close')]"))
                .click();

        System.out.println("Add New popup closed");
    }

    @Test(priority = 5, dependsOnMethods = "addNewUser")
    public void logoutTest() throws InterruptedException {

        driver.findElement(By.linkText("Logout"))
                .click();

        Thread.sleep(2000);

        System.out.println("Logout Successful");
    }

    @AfterTest
    public void closeBrowser() {

        driver.quit();
    }
}
