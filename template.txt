package PACKAGE_NAME;

import java.time.Duration;
import org.openqa.selenium.*;
import org.openqa.selenium.chrome.ChromeDriver;
import io.github.bonigarcia.wdm.WebDriverManager;
import org.testng.annotations.*;

public class CLASS_NAME {

    WebDriver driver;

    @BeforeClass
    public void setup() {
        WebDriverManager.chromedriver().setup();
        driver = new ChromeDriver();
        driver.manage().window().maximize();
        driver.manage().timeouts().pageLoadTimeout(Duration.ofSeconds(10));
        driver.get("BASE_URL");
        // Example: "http://localhost/student/admin"
    }

    @Test(priority = 1)
    public void adminLogin() throws InterruptedException {
        Thread.sleep(2000);
        driver.findElement(By.name("USERNAME_FIELD_NAME")).sendKeys("ADMIN_USERNAME");
        // Example field name: "username" | Example value: "admin"

        driver.findElement(By.name("PASSWORD_FIELD_NAME")).sendKeys("ADMIN_PASSWORD");
        // Example field name: "password" | Example value: "admin@123"

        driver.findElement(By.xpath("LOGIN_BUTTON_XPATH")).click();
        // Example: "//*[@id='login-frm']/button[1]"

        Thread.sleep(3000);
        System.out.println("Login done");
    }

    @Test(priority = 2)
    public void urlValidation() throws InterruptedException {
        driver.findElement(By.id("NAV_MENU_ITEM_ID")).click();
        // Example: "Employee" — the sidebar/nav link ID

        Thread.sleep(2000);
        String current = driver.getCurrentUrl();
        String expected = "EXPECTED_URL_AFTER_NAVIGATION";
        // Example: "http://localhost/student/admin/?page=students"

        System.out.println(current.equals(expected) ? "URL Valid" : "URL Invalid");
    }

    @Test(priority = 3)
    public void searchRecord() throws InterruptedException {
        driver.findElement(By.xpath("SEARCH_INPUT_XPATH")).sendKeys("SEARCH_KEYWORD");
        // Example xpath: "//*[@id='table_filter']/label/input"
        // Example keyword: "Ad"

        Thread.sleep(2000);
        System.out.println("Search done");
    }

    @Test(priority = 4)
    public void addRecord() throws InterruptedException {
        WebElement btn = driver.findElement(By.id("ADD_BUTTON_ID"));
        // Example: "new_std_btn"

        ((JavascriptExecutor) driver).executeScript("arguments[0].click();", btn);
        Thread.sleep(2000);

        // Fill form fields — add/remove as needed
        driver.findElement(By.name("FIELD_1_NAME")).sendKeys("FIELD_1_VALUE");
        // Example: By.name("firstname")  →  "Stanely"

        driver.findElement(By.name("FIELD_2_NAME")).sendKeys("FIELD_2_VALUE");
        // Example: By.name("lastname")   →  "Lambert"

        driver.findElement(By.name("FIELD_3_NAME")).sendKeys("FIELD_3_VALUE");
        // Example: By.name("grade")      →  "Grade 11"

        driver.findElement(By.name("FIELD_4_NAME")).sendKeys("FIELD_4_VALUE");
        // Example: By.name("phone")      →  "0777889090"

        driver.findElement(By.xpath("SAVE_BUTTON_XPATH")).click();
        // Example: "//button[contains(text(),'Save')]"

        Thread.sleep(2000);

        Alert alert = driver.switchTo().alert();
        System.out.println("Add Alert: " + alert.getText());
        alert.accept();
        Thread.sleep(2000);
    }

    @Test(priority = 5)
    public void deleteRecord() throws InterruptedException {
        driver.findElement(By.xpath("DELETE_BUTTON_XPATH")).click();
        // Example: "//*[@id='table']/tbody/tr[1]/td[7]/center/button[2]/i"
        // tr[1] = first row | button[2] = second button (delete)

        Thread.sleep(2000);

        Alert alert = driver.switchTo().alert();
        System.out.println("Delete Alert: " + alert.getText());
        alert.dismiss();  // dismiss = Cancel (do NOT delete)
        Thread.sleep(2000);
    }

    @Test(priority = 6)
    public void updateRecord() throws InterruptedException {
        driver.findElement(By.xpath("EDIT_BUTTON_XPATH")).click();
        // Example: "(//*[@id='table']/tbody/tr[1]/td[7]/center/button)[1]"
        // tr[1] = first row | button[1] = first button (edit)

        Thread.sleep(2000);

        WebElement fieldToUpdate = driver.findElement(By.name("FIELD_TO_UPDATE_NAME"));
        // Example: By.name("lastname")

        fieldToUpdate.clear();
        fieldToUpdate.sendKeys("UPDATED_VALUE");
        // Example: "User"

        driver.findElement(By.name("SAVE_UPDATE_BUTTON_NAME")).click();
        // Example: By.name("save")

        Thread.sleep(2000);

        Alert alert = driver.switchTo().alert();
        System.out.println("Update Alert: " + alert.getText());
        alert.accept();
        Thread.sleep(2000);
    }

    @Test(priority = 7)
    public void logout() {
        driver.findElement(By.xpath("LOGOUT_BUTTON_XPATH")).click();
        // Example: "/html/body/nav/div/div[2]/a"

        System.out.println("Logout done");
    }

    @AfterClass
    public void closeBrowser() {
        if (driver != null) driver.quit();
    }
}
